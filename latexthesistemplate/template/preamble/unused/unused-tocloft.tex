\IfPackageLoaded{subfigure}{\PassOptionsToPackage{subfigure}{tocloft}}

% apply style of TOC using tocloft
\IfPackageLoaded{tocloft}{
%%%% Layout TOC
\IfChapterDefined{%
	\renewcommand{\cftchappagefont}{\bfseries\fontTOC}  % Kapitel Seiten Schrift
	\renewcommand{\cftchapfont}{\bfseries\fontTOC}      % Kapitel Schrift
}
\renewcommand{\cftsecpagefont}{\fontTOC}            % Section Seiten Schrift
\renewcommand{\cftsubsecpagefont}{\fontTOC}         % Subsectin Seiten Schrift
\renewcommand{\cftsecfont}{\fontTOC}                % Section Schrift
\renewcommand{\cftsubsecfont}{\fontTOC}             % Subsection Schrift
}
